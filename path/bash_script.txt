#!/bin/bash
BASENAME=`basename "$PWD"`
VENDOR_PATH="$HOME/<yourpath>/$BASENAME/vendor";
DRUSH_PATH="$HOME/<yourpath>/$BASENAME/vendor/bin/drush";

if  [ -d $VENDOR_PATH ]
then
   eval  "$HOME/<yourpath>/$BASENAME/vendor/bin/drush"  "$@";
else
    echo  "This can only work if it is inside root directory of drupal";
fi
